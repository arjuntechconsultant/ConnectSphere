  <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r151/three.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/three@0.151.3/examples/js/controls/OrbitControls.min.js"></script>
  <script>
    // Scene setup
    const scene = new THREE.Scene();
    scene.background = new THREE.Color(0x202020);

    const camera = new THREE.PerspectiveCamera(
      75,
      window.innerWidth / window.innerHeight,
      0.1,
      1000
    );
    camera.position.set(0, 0, 15);

    const renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.body.appendChild(renderer.domElement);

    // Controls
    const controls = new THREE.OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;

    // Sample people nodes with positions
    const nodes = [
      { id: 1, name: "John", pos: [-4, 0, 0] },
      { id: 2, name: "James", pos: [0, 3, -1] },
      { id: 3, name: "Sara", pos: [3, -2, 1] },
      { id: 4, name: "Anna", pos: [-2, -3, 2] },
    ];

    // Connections between nodes (by id)
    const connections = [
      [1, 2], // John <-> James
      [2, 3], // James <-> Sara
      [1, 4], // John <-> Anna
    ];

    // Material for nodes (bubbles)
    const nodeMaterial = new THREE.MeshStandardMaterial({ color: 0x6699ff, roughness: 0.5, metalness: 0.7 });

    // Create spheres for each node
    const sphereGeometry = new THREE.SphereGeometry(0.7, 32, 32);
    const nodeMeshes = {};
    nodes.forEach(node => {
      const sphere = new THREE.Mesh(sphereGeometry, nodeMaterial.clone());
      sphere.position.set(...node.pos);
      sphere.userData = { name: node.name };
      scene.add(sphere);
      nodeMeshes[node.id] = sphere;
    });

    // Add light
    const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
    scene.add(ambientLight);
    const pointLight = new THREE.PointLight(0xffffff, 1);
    pointLight.position.set(10, 10, 10);
    scene.add(pointLight);

    // Create connection lines
    connections.forEach(([id1, id2]) => {
      const geometry = new THREE.BufferGeometry();
      const vertices = new Float32Array([
        ...nodeMeshes[id1].position.toArray(),
        ...nodeMeshes[id2].position.toArray()
      ]);
      geometry.setAttribute('position', new THREE.BufferAttribute(vertices, 3));
      const material = new THREE.LineBasicMaterial({ color: 0xffffff });
      const line = new THREE.Line(geometry, material);
      scene.add(line);
    });

    // Animate function
    function animate() {
      requestAnimationFrame(animate);
      controls.update();
      renderer.render(scene, camera);
    }

    animate();

    window.addEventListener('resize', () => {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    });
  </script>
